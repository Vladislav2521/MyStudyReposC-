void Sum(params int[] numbers)
{
    int result = 0;
    foreach (var n in numbers)
    {
       result += n;    
    }
    Console.WriteLine(result);
}

int[] nums = { 1, 2, 3, 4, 5 };
Sum(nums);
Sum(1, 2, 3, 4);
Sum(1, 2, 3);
Sum();

int Sum(params int[] parametres) // Все параметры, которые мы задали помещаются в "parametres".
// В цикле мы перебираем элементы массива и в переменную "result" значение каждого элемента, итерируясь по циклу, добавляем.
// После того, как мы добавили все элементы, результат у нас возвращается при помощи "return result" и мы получаем его в переменной "SumResult".
{
    int result = 0; // Создаем переменную куда будем складывать результат.

	for (int i = 0; i < parametres.Length; i++) // Перебираем все элементы массива при помощи цикла и "parametres.Length".
	{
		result += parametres[i]; // Значение из каждого элемента массива прибавляем в переменную "result".
	}
	return result; // Затем переменную "result" возвращаем в качестве результата работы этого метода.
}
// Ключевое слово "params" используем чтобы передать каждый параметр отдельно через запятую и количество этих параметров могло быть разным.

int SumResult = Sum(5,10,9,7); // Тут мы можем передавать внутрь любое количество параметров, а можем вообще их не передать.
					   // Если мы не передадим никаких параметров, то это будет означать, что в этом массиве у нас вообще не будет никаких элементов.
					   // В этом случае они не будут добавлены в переменную "result". там как был 0, так и останется.
Console.WriteLine(SumResult);
